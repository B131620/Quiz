Setup steps:
1. Open your Google Sheet → Extensions → Apps Script
2. Replace all code with the script below
3. Click Deploy → New Deployment → Web App
4. Set "Who has access" → Anyone
5. Copy the Web App URL → paste in quiz setup 

function doPost(e) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheet = ss.getActiveSheet();
    var data = JSON.parse(e.postData.contents);

    if (sheet.getLastRow() === 0) {
      var headers = [
        'Timestamp','Name','Email','Score','Max Score','%','Grade',
        'Correct','Wrong','Skipped','Neg Deducted',
        'Radio Score','Radio Correct','Radio Wrong',
        'TF Score','TF Correct','TF Wrong',
        'Multi Score','Multi Correct','Multi Wrong',
        'Code Score','Code Correct','Code Wrong'
      ];
      for (var i = 1; i <= 30; i++) {
        headers.push('Q'+i+' Result', 'Q'+i+' Pts', 'Q'+i+' Time(s)');
      }
      sheet.appendRow(headers);
      sheet.getRange(1,1,1,headers.length)
        .setFontWeight('bold').setBackground('#1e1a2e').setFontColor('#ffffff');
      sheet.setFrozenRows(1);
    }

    var row = [
      data.timestamp, data.name, data.email,
      data.score, data.maxScore, data.pct+'%', data.grade,
      data.correct, data.wrong, data.skipped, data.negDed,
      data.byType.radio.score, data.byType.radio.correct, data.byType.radio.wrong,
      data.byType.tf.score, data.byType.tf.correct, data.byType.tf.wrong,
      data.byType.multi.score, data.byType.multi.correct, data.byType.multi.wrong,
      data.byType.code.score, data.byType.code.correct, data.byType.code.wrong
    ];
    data.qs.forEach(function(q){ row.push(q.result, q.pts, q.time); });
    sheet.appendRow(row);

    var lr = sheet.getLastRow();
    var p = parseFloat(data.pct);
    var bg = p>=75?'#d1fae5':p>=50?'#fef9c3':'#fee2e2';
    sheet.getRange(lr,6).setBackground(bg);

    return ContentService
      .createTextOutput(JSON.stringify({ok:true}))
      .setMimeType(ContentService.MimeType.JSON);
  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ok:false,err:err.message}))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
